\set userName postgres
\set hostName localhost
\set portNumber 5432


-------------
-- Nome da BD
-------------
\set dataBase my_gis_raster
;

-------------------------
-- Remover e Criar BD GIS
-------------------------
\echo
\echo "Remover Base de Dados" :dataBase
;
DROP DATABASE IF EXISTS :dataBase
;


\echo
\echo "Criar Base de Dados" :dataBase
;
CREATE DATABASE :dataBase
;


\echo
\echo "Estabelecer Conexao com a Base de Dados" :dataBase
;
\c :dataBase :userName :hostName :portNumber
;


\echo
\echo "Aplicar o Extensor "postgis" à Base de Dados" :dataBase
;
CREATE EXTENSION postgis
;
-- (nov.2023)
-- proxima linha foi adicionada pois o tipo "raster"
-- nao esta diretamente disponivel na nova versao do extensor "postgis"

;
